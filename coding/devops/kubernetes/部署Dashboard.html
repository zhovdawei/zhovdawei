# 部署Dashboard

# 翻墙安装

如果你能流畅访问外网，你可以直接执行下面的命令

$ kubectl apply -f https://raw.githubusercontent.com/kubernetes/dashboard/v1.10.1/src/deploy/recommended/kubernetes-dashboard.yaml
1

# 非翻墙安装

# 下载kubernetes-dashboard.yaml

$ wget https://raw.githubusercontent.com/kubernetes/dashboard/v1.10.1/src/deploy/recommended/kubernetes-dashboard.yaml
1

# 修改kubernetes-dashboard.yaml参数

$ vi kubernetes-dashboard.yaml

# ------------------- Dashboard Deployment ------------------- #
        # 更换地址
        image: lizhenliang/kubernetes-dashboard-amd64:v1.10.1
        
# ------------------- Dashboard Service ------------------- #
kind: Service
apiVersion: v1
metadata:
  labels:
    k8s-app: kubernetes-dashboard
  name: kubernetes-dashboard
  namespace: kube-system
spec:
  # 暴露方式nodeport
  type: NodePort
  ports:
    - port: 443
      targetPort: 8443
      # 暴露端口 30001
      nodePort: 30001
  selector:
    k8s-app: kubernetes-dashboard

# 创建用户

# 创建 名为：dashboard-admin 的serviceaccount
[root@k8s-master ~]# kubectl create serviceaccount dashboard-admin -n kube-system
serviceaccount/dashboard-admin created

# 将dashboard-admin 绑定在 kube-system:dashboard-admin 权限组下
# kube-system:dashboard-admin 中的dashboard-admin是一个权限组，拥有k8s的全部权限，相当于超级管理员
[root@k8s-master ~]# kubectl create clusterrolebinding dashboard-admin --clusterrole=cluster-admin --serviceaccount=kube-system:dashboard-admin
clusterrolebinding.rbac.authorization.k8s.io/dashboard-admin created

# 拿到账号的权限认证的token
