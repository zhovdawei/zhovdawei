# 镜像的层次


基础环境层 : 指的就是操作系统层面：centos,debian,ubuntu
运行环境层 : tomcat,jdk
项目层 : war包，jar包
可以理解为，对通用镜像的抽象。个性化的镜像，依赖通用镜像完成。

###################################################################
# 依赖一个运行镜像tomcat,它是运行war包的容器
# 当然在这个运行镜像内，依赖一个操作系统镜像，这就是镜像的依赖关系
FROM lizhenliang/tomcat 
# 标记 镜像的作者
LABEL maintainer www.ctnrs.com
# 删除镜像中tomcat/webapps/目录中的文件
RUN rm -rf /usr/local/tomcat/webapps/*
# 上面mvn编译完，产生一个target目录，里面有一个war包，将它加入到镜像webapps/目录下
ADD target/*.war /usr/local/tomcat/webapps/ROOT.war 
###################################################################

# 配置和运行pod

这里是发布java-web应用属于无状态应用，所以选择deployment

# 生成一个标准的yaml
$ kubectl create deployment java-demo --image=david0607/java-demo --dry-run -o yaml > deploy.yaml

# 编辑yaml
$ vi deploy.yaml
########################################################
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: java-demo
  name: java-demo
spec:
  # 副本数
  replicas: 3
  selector:
    matchLabels:
      app: java-demo
  template:
    metadata:
      creationTimestamp: null
      labels:
        app: java-demo
    spec:
      containers:
      # 镜像仓库地址(docker hub 公共的)
      - image: david0607/java-demo
        name: java-demo
########################################################

# 运行pod,pod里面跑docker容器，根据我们配置的镜像地址
# 对于k8s,最小运行单元是pod
# 通过我们配置的yaml去起pod
$ kubectl apply -f deploy.yaml

# 查看pod,这里副本启动的速度跟你容器下载镜像的速度有关，还是自建仓库快
[root@k8s-master ~]# kubectl get pods

# 暴露应用

service管理着pod的关联与访问

# 生成一个标准的service yaml文件
# --port=80 service的端口，集群内部访问用
# --target-port=8080 容器里面的跑的服务的端口，tomcat默认就是8080
# --type=NodePort 会随机生成一个端口，供集群外部用户访问用，这个端口可以指定具体的。
[root@k8s-master ~]# kubectl expose deployment java-demo --port=80 --target-port=8080 --type=NodePort -o yaml --dry-run > svc.yaml

# 编辑service yaml文件
$ vi svc.yaml
###############################
apiVersion: v1
kind: Service
metadata:
  labels:
    app: java-demo
  name: java-demo
spec:
  ports:
  - port: 80
    protocol: TCP
    targetPort: 8080
  selector:
    app: java-demo
  type: NodePort
###############################

# 运行
[root@k8s-master ~]# kubectl apply -f svc.yaml 

# 查看pod和svc
