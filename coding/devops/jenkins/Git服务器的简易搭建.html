# Git服务器的简易搭建

# 准备Git服务器

我这里用192.168.163.105做Git服务器

# 1.安装Git

[root@host105 ~]# yum install git -y
1

# 2.清一下环境规则

[root@host105 ~]# iptables -F
1

# 3.创建一个用户

[root@host105 ~]# useradd git
[root@host105 ~]# passwd git
Changing password for user git.
New password: 
BAD PASSWORD: The password fails the dictionary check - it is based on a dictionary word
Retype new password: 
passwd: all authentication tokens updated successfully.

# 4.进入Git

[root@host105 ~]# su - git
[git@host105 ~]$

# 5.创建Git仓库

[git@host105 ~]$ mkdir repos
1

# 6.在仓库目录里面创建一个项目目录

[git@host105 repos]$ mkdir app.git
[git@host105 app.git]$ pwd

# 9.将上面的秘钥放入Git服务器中

# 切回105机
[root@host105 ~]# su - git
[git@host105 ~]$ ls -al
total 16
drwx------. 4 git  git  108 Dec  1 00:08 .
drwxr-xr-x. 4 root root  30 Nov 30 18:55 ..
-rw-------. 1 git  git  216 Dec  1 00:08 .bash_history
-rw-r--r--. 1 git  git   18 Aug  8 20:06 .bash_logout
-rw-r--r--. 1 git  git  193 Aug  8 20:06 .bash_profile
-rw-r--r--. 1 git  git  231 Aug  8 20:06 .bashrc
drwxrwxr-x. 3 git  git   21 Nov 30 19:07 repos
drwx------. 2 git  git   25 Nov 30 19:20 .ssh

# 如果不存在 .ssh 请先创建
[git@host105 ~]$ chmod 700 .ssh/

# 创建文件authorized_keys，并把上一步的秘钥串复制粘贴其中
[git@host105 ~]$ vi .ssh/authorized_keys

# 修改这个文件的权限 
[git@host105 ~]$ chmod 600 .ssh/authorized_keys
