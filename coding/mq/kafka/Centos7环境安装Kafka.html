# Centos7环境安装Kafka

# 下载安装包

官方下载地址 (opens new window) 我这里选择的版本为:kafka_2.13-2.4.0.tgz

# 修改配置文件

[root@Host201 kafka_2.13-2.4.0]# vim config/server.properties 

# 该实例在集群的中唯一标识，建议取值 > 0
broker.id=1

# 实例监听地址(kafka实例与其他应用交互的地址)
listeners=PLAINTEXT://192.168.163.201:9092

# 日志存储地址
log.dirs=/tmp/kafka-logs

# 连接zookeeper，kafka集群由zk调度
zookeeper.connect=192.168.163.201:2181

# 其他配置视需求而定

# 启动测试

# 注意：启动kafka前，请务必确保zookeeper已启动
# -daemon 由守护线程后台运行
[root@Host201 kafka_2.13-2.4.0]# bin/kafka-server-start.sh -daemon  config/server.properties

# 验证kafka启动成功
[root@Host201 kafka_2.13-2.4.0]# jps
4880 Jps
2241 Elasticsearch
2840 QuorumPeerMain
4809 Kafka

# 创建topic
bin/kafka-topics.sh --create --zookeeper 192.168.163.201:2181 --replication-factor 1 --partitions 1 --topic test

# 查看topic
bin/kafka-topics.sh --list --zookeeper 192.168.163.201:2181
