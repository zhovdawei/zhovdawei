    # 查看rabbitmq后台管理页面

    浏览器输入192.168.163.102:15672
    192.168.163.102是我本机的ip,你应该输入自己的。
    15672是rabbitmq对外暴露的后台端口。
    使用rabbitmq默认的管理员账户密码登录：guest/guest

    # rabbitmq启动慢，后台页面加载慢

    # 修改network文件
[root@host102 ~]# vi /etc/sysconfig/network
# 加入下面这行参数
# host102是本机的名字
HOSTNAME=host102

# 修改hosts文件
[root@host102 ~]# vi /etc/hosts
# 加入下面这行参数
# ip为本机绑定ip
# host102为本机名
192.168.163.102 host102

# 重启rabbitmq
[root@host102 ~]# service rabbitmq-server restart
