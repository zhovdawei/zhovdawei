# 修改配置文件

[root@host114 kibana-7.5.0-linux-x86_64]# vi config/kibana.yml 
###############################################################
# 为Kibana远程访问绑定IP
server.host: 192.168.163.114
# 绑定elasticsearch集群，这里先设置一个测试
elasticsearch.hosts: ["http://192.168.163.114:9200"]
###############################################################

# Kibana配置和使用

# 配置文件

# 重要配置说明
# 远程访问地址绑定
server.host: 192.168.163.114

# 访问端口 除非5601被占用，默认是5601
server.port: 5601

# 绑定elasticsearch集群
elasticsearch.hosts: ["http://192.168.163.114:9200"]
